
# cut video from start to end time
ffmpeg -ss 00:01:00 -to 00:02:00 -i input.mp4 -c copy output.mp4

# crop video X-Y
# x_size:y_size:x_offset:y_offset

# a way faster command
ffmpeg -i source.mkv -filter:v "crop=459:632:340:39" -c:a copy out.mp4

# a slow one just in case
ffmpeg -i input.mp4 -vf "crop=459:632:340:39" -c:v libx265 -crf 17 -c:a copy ouptut.mp4


#resize video
#    scale=widht:height
#    -1 means that you specify height/widht and the second parameter will be calculated automatically to maintain aspect ratio
#    veryslow means good compression but takes more time
ffmpeg -i MyMovie.mkv -vf scale=-1:720 -c:v libx264 -crf 18 -preset veryslow -c:a copy MyMovie_720p.mkv
